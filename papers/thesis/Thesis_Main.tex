% INCLUDE-CHAPTER MACRO
% Increments chapter counter, resets subordinate counters, adds to TOC
% ===================================================================
\newcommand{\includechapter}[2]{%  % #1 = TOC title, #2 = file path
  \cleardoublepage
  \refstepcounter{chapter}%
  \chaptermark{#1}%
  \setcounter{section}{0}%
  \setcounter{equation}{0}%
  \setcounter{figure}{0}%
  \setcounter{table}{0}%
  \setcounter{theorem}{0}%
  \setcounter{footnote}{0}%
  \addcontentsline{toc}{chapter}{\protect\numberline{\thechapter}#1}%
  {\huge\bfseries\noindent Chapter~\thechapter\\[6pt] #1\par}%
  \vspace{1cm}%
  \input{#2}%
}
